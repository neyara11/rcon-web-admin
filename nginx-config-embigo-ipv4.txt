# Конфигурация nginx для rcon-web-admin с SSL и доменом embigo.ru
# Приложение работает по пути /rwa, веб-сокет по пути /rwa-ws
# Используем 127.0.1 вместо localhost для избежания проблем с IPv6

server {
    listen 80;
    server_name embigo.ru;
    # Перенаправление с HTTP на HTTPS
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name embigo.ru;

    # Пути к SSL-сертификатам
    ssl_certificate /path/to/your/certificate.crt; # Замените на путь к вашему SSL-сертификату
    ssl_certificate_key /path/to/your/private.key; # Замените на путь к вашему приватному ключу

    # Настройки SSL
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-RSA-AES256-GCM-SHA512:DHE-RSA-AES256-GCM-SHA512:ECDHE-RSA-AES256-GCM-SHA384:DHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers off;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;

    # Заголовки безопасности
    add_header X-Frame-Options DENY;
    add_header X-Content-Type-Options nosniff;
    add_header X-XSS-Protection "1; mode=block";
    add_header Strict-Transport-Security "max-age=6307200; includeSubDomains; preload";

    # Прокси-настройки для rcon-web-admin
    location /rwa/ {
        proxy_pass http://127.0.0.1:4326/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_redirect off;
        # Передаем информацию о префиксе URL
        proxy_set_header X-Script-Name /rwa;
    }

    # Прокси-настройки для WebSocket rcon-web-admin
    location ~ ^/rwa-ws {
        proxy_pass http://127.0.0.1:4327;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
